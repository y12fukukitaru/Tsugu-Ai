-- ============================================================ -- Tsugu Phase 11 追加SQL -- 内容: 認定FDEから顧客への「提案メモ」テーブル fde_notes -- 前提: Phase 1〜10 のSQLを実行済み（is_my_client / is_admin を利用） -- 使い方: Supabase → SQL Editor に貼り付けて Run（翻訳はオフに） -- ============================================================ create table if not exists public.fde_notes ( id uuid primary key default gen_random_uuid(), customer_id uuid not null references auth.users(id) on delete cascade, body text not null, created_by uuid, created_at timestamptz not null default now() ); create index if not exists idx_fdenotes_customer on public.fde_notes(customer_id); alter table public.fde_notes enable row level security; -- 顧客は自分宛のメモを閲覧、担当認定FDEと運営は閲覧できる drop policy if exists "fdenotes_select" on public.fde_notes; create policy "fdenotes_select" on public.fde_notes for select using (customer_id = auth.uid() or public.is_my_client(customer_id) or public.is_admin()); -- 書き込みは担当認定FDE（または運営）のみ drop policy if exists "fdenotes_insert" on public.fde_notes; create policy "fdenotes_insert" on public.fde_notes for insert with check (created_by = auth.uid() and (public.is_my_client(customer_id) or public.is_admin())); drop policy if exists "fdenotes_delete" on public.fde_notes; create policy "fdenotes_delete" on public.fde_notes for delete using (created_by = auth.uid() or public.is_admin());
